template<class T>
void BinaryInsertionSort(T data[], int n) {
    int left, right, mid;
    for (int i = 1; i < n; i++) {
        T temp = data[i];
        left = 0, right = i - 1;
        while (left <= right) {
            mid = (left + right) / 2;
            if (data[mid] > temp)
                right = mid - 1;
            else
                left = mid + 1;
        }
        for (int j = i - 1; j >= left; j--)
            data[j + 1] = data[j];
        data[left] = temp;
    }
}
